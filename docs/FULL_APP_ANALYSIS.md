## Техническая архитектура для новых функций

### Новые таблицы БД

```sql
-- Подписки
CREATE TABLE subscriptions (
  id SERIAL PRIMARY KEY,
  telegram_id BIGINT REFERENCES users(telegram_id),
  plan VARCHAR(50) NOT NULL,
  status VARCHAR(50) NOT NULL,
  starts_at TIMESTAMP,
  expires_at TIMESTAMP,
  payment_method VARCHAR(50),
  created_at TIMESTAMP DEFAULT NOW()
);

-- Друзья
CREATE TABLE friendships (
  id SERIAL PRIMARY KEY,
  user1_id BIGINT REFERENCES users(telegram_id),
  user2_id BIGINT REFERENCES users(telegram_id),
  status VARCHAR(50) DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT NOW()
);

-- Комментарии
CREATE TABLE comments (
  id SERIAL PRIMARY KEY,
  demo_id VARCHAR(100) NOT NULL,
  telegram_id BIGINT REFERENCES users(telegram_id),
  content TEXT NOT NULL,
  parent_id INTEGER REFERENCES comments(id),
  likes INTEGER DEFAULT 0,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Спины колеса
CREATE TABLE wheel_spins (
  id SERIAL PRIMARY KEY,
  telegram_id BIGINT REFERENCES users(telegram_id),
  prize_type VARCHAR(50) NOT NULL,
  prize_amount INTEGER,
  spin_date DATE DEFAULT CURRENT_DATE,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Квесты
CREATE TABLE quest_progress (
  id SERIAL PRIMARY KEY,
  telegram_id BIGINT REFERENCES users(telegram_id),
  quest_id VARCHAR(100) NOT NULL,
  current_step INTEGER DEFAULT 0,
  completed BOOLEAN DEFAULT FALSE,
  started_at TIMESTAMP DEFAULT NOW(),
  completed_at TIMESTAMP
);

-- Battle Pass
CREATE TABLE battle_pass (
  id SERIAL PRIMARY KEY,
  telegram_id BIGINT REFERENCES users(telegram_id),
  season INTEGER NOT NULL,
  level INTEGER DEFAULT 0,
  xp INTEGER DEFAULT 0,
  is_premium BOOLEAN DEFAULT FALSE,
  claimed_rewards JSONB DEFAULT '[]',
  created_at TIMESTAMP DEFAULT NOW()
);
```
